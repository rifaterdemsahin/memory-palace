⚡ Quick Reference — Qdrant + Ollama

# Pull embedding model
ollama pull nomic-embed-text

# Start containers
docker compose up -d

# Check Qdrant health
curl http://localhost:6333/healthz

# Create collection (768 dims for nomic-embed-text)
curl -X PUT http://localhost:6333/collections/memory_palace \
  -H 'Content-Type: application/json' \
  -d '{"vectors": {"size": 768, "distance": "Cosine"}}'

# Embed a document
curl http://localhost:11434/api/embeddings \
  -d '{"model":"nomic-embed-text","prompt":"Your text here"}'

🎮 Quick Reference — Roblox Luau

-- Basic memory locus script in Roblox Studio
local locus = script.Parent
local info = locus:WaitForChild("InfoLabel")

locus.Touched:Connect(function(part)
  local player = game.Players:GetPlayerFromCharacter(part.Parent)
  if player then
    info.Text = "💡 " .. locus.Name .. ": Concept description here"
    info.Visible = true
    task.delay(5, function() info.Visible = false end)
  end
end)
