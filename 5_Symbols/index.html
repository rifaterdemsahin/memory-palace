🎮 Roblox Luau — Memory Locus Script

-- MemoryLocus.lua
-- Place this as a Script inside a Part in Roblox Studio
-- The part becomes a "locus" — a knowledge anchor in the memory palace

local RunService = game:GetService("RunService")
local Players = game:GetService("Players")

local locus = script.Parent
local concept = locus:GetAttribute("Concept") or locus.Name
local detail  = locus:GetAttribute("Detail") or "No detail set"
local tags    = locus:GetAttribute("Tags") or ""

-- Proximity prompt (Roblox built-in)
local prompt = Instance.new("ProximityPrompt")
prompt.ActionText = "Learn"
prompt.ObjectText = concept
prompt.MaxActivationDistance = 8
prompt.Parent = locus

prompt.Triggered:Connect(function(player)
  -- Fire RemoteEvent to show GUI on client
  local remotes = game.ReplicatedStorage:FindFirstChild("MemoryPalaceEvents")
  if remotes then
    remotes.ShowLocus:FireClient(player, {
      concept = concept,
      detail  = detail,
      tags    = tags,
    })
  end
end)

🐍 Python — Embedding Pipeline

"""
embed_documents.py
Reads all markdown files and embeds them into Qdrant via Ollama.
Run: python embed_documents.py
"""
import os, json, pathlib, requests

OLLAMA_URL  = os.getenv("OLLAMA_HOST", "http://localhost:11434")
QDRANT_URL  = os.getenv("QDRANT_HOST", "http://localhost:6333")
MODEL       = os.getenv("OLLAMA_EMBED_MODEL", "nomic-embed-text")
COLLECTION  = os.getenv("QDRANT_COLLECTION", "memory_palace")

def embed(text: str) -> list[float]:
    r = requests.post(f"{OLLAMA_URL}/api/embeddings",
                      json={"model": MODEL, "prompt": text})
    r.raise_for_status()
    return r.json()["embedding"]

def upsert(point_id: int, vector: list[float], payload: dict):
    r = requests.put(f"{QDRANT_URL}/collections/{COLLECTION}/points",
                     json={"points": [{"id": point_id, "vector": vector, "payload": payload}]})
    r.raise_for_status()

def main():
    md_files = list(pathlib.Path(".").rglob("*.md"))
    for i, path in enumerate(md_files):
        if "obsolete" in str(path): continue
        text = path.read_text(encoding="utf-8")
        print(f"Embedding {path} ...")
        vector = embed(text[:4096])  # nomic-embed-text max context
        upsert(i, vector, {"file": str(path), "section": path.parts[0]})
    print(f"✅ Embedded {len(md_files)} documents into Qdrant")

if __name__ == "__main__":
    main()
